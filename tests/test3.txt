COMPILER Aritmetica    

CHARACTERS 

upletter = 'A' .. 'Z' .
downletter = 'a' .. 'z' .
letter = "abcdefghijklmnopqrstuvwxyz" - upletter - downletter. 
digit = "0123456789" .
hexdigit = digit + "ABCDEF".

KEYWORDS

while = "while".
do = "do".
if = "if".
switch = "switch".

TOKENS

digit = digit.
ident = letter {letter | digit} EXCEPT KEYWORDS .
number = digit {digit} .
signnumber = [sign] digit {digit} .

END Aritmetica
